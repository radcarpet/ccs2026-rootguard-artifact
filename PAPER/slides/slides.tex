\section{Introduction}
% -------------------------------------------------

\begin{frame}{Introduction}
  \begin{itemize}
    \item \textbf{Setting}: Multi-turn conversations between a user and an untrusted remote LLM, that requests sensitive information to complete a task (e.g. medical diagnosis).
    \item \textbf{Problem}: The remote LLM can request private values that have already been released, or new private values that are a function of previously released ones.
    \item \textbf{Goal}: Reduce the expected total privacy leakage/budget expenditure over several conversations.
    \item \textbf{Method}: Maintain a dependency graph for all private values derived, released or obtained (from the remote LLM).
    \item \textbf{Metric}: How do we compare leakage?
  \end{itemize}
\end{frame}

% -------------------------------------------------
\section{Definitions}
% -------------------------------------------------

\begin{frame}{Definitions}
\begin{block}{Remote LLM $\mathcal{A}$}
    A remote LLM takes in a task $t$ requiring a set of (private) inputs $\{x_1,\dots,x_n\}$ and returns a response $r = \mathcal{A}(t, \{x_i\}^n_{i=1})$.
\end{block}
\begin{block}{Quality Oracle $\mathcal{Q}$}
    A quality oracle evaluates the goodness of a remote LLM response $r$, for a task $t$ as $\mathcal{Q}(t, r)$
\end{block}
\begin{block}{$\gamma$-Sensitive Task $t_\gamma$}
    Given a remote LLM $\mathcal{A}$, a $\gamma$-sensitive task $t_\gamma$ satisfies
    \[
    \mathcal{Q}\big(t_\gamma, \mathcal{A}(t_\gamma, \{x_i\}^n_{i=1})\big) = \mathcal{Q}\big(t_\gamma, \mathcal{A}(t_\gamma, \{x'_i\}^n_{i=1})\big)
    \]
    for a set of (private) inputs $\{x_i\}^n_{i=1}$ and $\{x'_i\}^n_{i=1}$ \textbf{iff}, $d_{\mathcal{X}}\big(x_i, x'_i\big) \le \gamma$ for all $i \in \{1,\dots,n\}$.
  \end{block}
\end{frame}


% -------------------------------------------------
\section{Evaluation}
% -------------------------------------------------

\begin{frame}{Eval Setup - 1}
We set a required privacy guarantee for each private value being released (say $\epsilon$). We have an advantage when $\mathcal{A}$ requests a previously released value or a function of a previously released values. \\

\begin{enumerate}
    \item \textbf{Vanilla Preempt}:
    \begin{itemize}
        \item $\mathbf{R_1}: x\xrightarrow{\epsilon}x'$; $\mathbf{R_2}: x\xrightarrow{\epsilon}x'$. Loss $\propto 2\epsilon$ 
        \item $\mathbf{R_1}: x\xrightarrow{\epsilon}x'$; $\mathbf{R_2}: f(x)\xrightarrow{\epsilon}f(x)'$. Loss $\propto 2\epsilon$ 
    \end{itemize}
    \item \textbf{Dependency Preserving Preempt}:
    \begin{itemize}
        \item $\mathbf{R_1}: x\xrightarrow{\epsilon}x'$; $\mathbf{R_2}: x'\xrightarrow{0}x'$. Loss $\propto \epsilon$ 
        \item $\mathbf{R_1}: x\xrightarrow{\epsilon}x'$; $\mathbf{R_2}: f(x')\xrightarrow{\epsilon - \frac{\epsilon}{m}}f(x')'$. Loss $\propto 2\epsilon - \frac{\epsilon}{m}$ 
    \end{itemize}
\end{enumerate}


    \begin{block}{Postprocessing Property - Sarthak}
        Let $(\mathcal{X}, d_{\mathcal{X}})$ and $(\mathcal{Y}, d_{\mathcal{Y}})$ be metric spaces, let $\mathcal{M}: \mathcal{X} \to \mathcal{X}$ be a $\epsilon$-mDP mechanism on $(\mathcal{X}, d_{\mathcal{X}})$ and $f: \mathcal{X} \to \mathcal{Y}$ be a bi-Lipschitz function constants $m, L$. Then $\phi = f \circ \mathcal{M}$ is  $\frac{\epsilon}{m}$-mDP in $(\mathcal{Y}, d_{\mathcal{Y}})$.
    \end{block}

\end{frame}

\begin{frame}{Eval Setup - 2}
    We can also frame this in terms of the utility guarantee. Consider a $t_\gamma$ task, where the mechanism $\mathcal{M}$ must satisfy $d_\mathcal{X}(x_i, \mathcal{M}(x_i)) \le \gamma$ for all private values $x_i$ in a conversation, for the utility guarantee. \\
    \vspace{5mm}
    Can we use this to bound the noise added to any sensitive value as part of the DP-guarantee?

    \[
    \Pr[\mathcal{M}(x) \in S]
    \;\leq\;
    e^{\epsilon \, d_{\mathcal{X}}(x, x')}
    \Pr[\mathcal{M}(x') \in S],
    \]

    $d_{\mathcal{X}}(x, x')$ represents a worst-case distance between any pair of values in $\mathcal{X}$, and the actual budget used is $\epsilon \times d_{\mathcal{X}}(x, x')$. Bounding $ d_{\mathcal{X}}(x, x')$ to $\gamma$ lets us compare $\epsilon$ factors easily.
    
\end{frame}

\begin{frame}{Data}
    \textbf{Synthetic multi-turn conversations for medical/financial attributes.} 
    
    The actual task is not relevant here, since we want to measure privacy loss per conversation. As long as the private values satisfy the $t_\gamma$ condition, our utility guarantee is met. \\
    
    \vspace{5mm}
    
    \textbf{MIMIC-IV}. 
    
    We can use Mimic's independent medical attributes in place of the synthetic ones for a ``real-world" dataset, and compute derived values as before.
    
\end{frame}
